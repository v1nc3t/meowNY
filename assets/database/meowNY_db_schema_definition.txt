// --- Enumerations ---

Enum transaction_type {
  INCOME
  EXPENSE
}

Enum frequency_type {
  DAILY
  WEEKLY
  BIWEEKLY
  TRIWEEKLY
  MONTHLY
  BIMONTHLY
  QUARTERLY
  YEARLY
}

// --- Tables ---

Table users {
  id bigint [pk, increment]
  firstname varchar(50) [not null]
  lastname varchar(30) [not null]
  email varchar(254) [unique, not null]
  username varchar(30) [not null]
  password varchar(100) [not null]
  created_at timestamp [not null]
  updated_at timestamp [not null]
}

Table categories {
  id bigint [pk, increment]
  user_id bigint [not null]
  type transaction_type [not null]
  name varchar(50) [not null]
  created_at timestamp [not null]
  updated_at timestamp [not null]

  Indexes {
    user_id [name: "idx_category_user"]
    (user_id, type, name) [unique, name: "unique_user_category_type_name"]
  }
}

Table budgets {
  id bigint [pk, increment]
  user_id bigint [not null]
  category_id bigint [not null]
  amount_limit numeric(12,2) [not null]
  month integer [not null]
  year integer [not null]
  created_at timestamp [not null]
  updated_at timestamp [not null]

  Indexes {
    (user_id, year, month) [name: "idx_budget_user_period"]
    (user_id, category_id) [name: "idx_budget_user_category"]
    (user_id, category_id, month, year) [unique, name: "unique_user_category_period"]
  }
}

Table recurring_transactions {
  id bigint [pk, increment]
  user_id bigint [not null]
  category_id bigint [not null]
  type transaction_type [not null]
  name varchar(50) [not null]
  amount numeric(12,2) [not null]
  next_due_date date [not null]
  frequency frequency_type [not null]
  is_active boolean [not null, default: true]
  created_at timestamp [not null]
  updated_at timestamp [not null]

  Indexes {
    (is_active, next_due_date) [name: "idx_recurring_tx_worker"]
    user_id [name: "idx_recurring_tx_user"]
  }
}

Table transactions {
  id bigint [pk, increment]
  user_id bigint [not null]
  category_id bigint [not null]
  recurring_transaction_id bigint [null]
  type transaction_type [not null]
  name varchar(50) [not null]
  amount numeric(12,2) [not null]
  payment_date date [not null]
  description varchar(100) [null]
  created_at timestamp [not null]
  updated_at timestamp [not null]

  Indexes {
    (user_id, type, payment_date) [name: "idx_tx_user_type_date"]
    category_id [name: "idx_tx_category"]
    recurring_transaction_id [name: "idx_tx_template"]
  }
}

// --- Relationships & Foreign Keys ---

Ref: categories.user_id > users.id [delete: cascade]
Ref: budgets.user_id > users.id [delete: cascade]
Ref: budgets.category_id > categories.id [delete: cascade]
Ref: recurring_transactions.user_id > users.id [delete: cascade]
Ref: recurring_transactions.category_id > categories.id [delete: restrict]
Ref: transactions.user_id > users.id [delete: cascade]
Ref: transactions.category_id > categories.id [delete: restrict]
Ref: transactions.recurring_transaction_id > recurring_transactions.id [delete: set null]
